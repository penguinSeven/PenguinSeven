for i=0, a

   if i > (a/2) then

        str = string.rep("*", 2*i + 1)
   end

   print(space .. str)
end
  • function 由 C 或 lua 编写的函数
print(type(type)) --> function

-- ---------
-- 可变参数
-- ---------

function add(...)
  local s = 0
  for i,v in ipairs{...}
  do
          s = s + i
  end
  return s
end


print(add(4,5,6,7,8))


-- --------------
-- 最大值，多值返回
-- --------------

function maxValue(a)
  local maxK = 1
  local maxV = a[maxK]

  for k,v in ipairs(a)
  do
      if v > maxV
      then
          maxK = k
          maxV = v
      end
  end

  return maxV,maxK
end

# 变量以及作用域

Lua 变量有三种类型：全局变量、局部变量、表中的域。

  • 作用域

a = 5           -- 全局变量
local b = 6     -- 局部变量

function joke()
    c = 7       -- 全局变量
    local d = 8 -- 局部变量
end

joke()
print(c,d)      -- 7,nil


do
    local a =6  -- 局部变量
    b = 5       -- 对全局变量重新赋值
    print(a,b)  -- 输出 6,5
end

print(a,b)      -- 输出 5,5
  • 赋值语句
-- 拼接语句
a = "hello" .. " " .. " word"

-- 多个变量同时赋值
a,b = 0,1

-- 个数不匹配时
a,b = 1     <--> a=1;b=nil
a,b = 1,2,3   <--> a=1;b=2 多余的被忽略
  • 索引

对 table 的索引使用方括号 []。Lua 也提供了 . 操作。

site = {}
site["key"] = "www.baidu.com"

print(site['key'])
print(site.key)

  • table 默认下表从1开始

for k,v in pairs(table1) do
    print(k .. "-" .. v)
end
  • function 支持存储在变量里
-- --------
-- 匿名赋值变量
-- --------
function facto(n)
    if n == 0 then
        return 1
    else
        return n * facto(n-1)
    end
end

print(facto(5))
facto2 = facto
print(facto2(6))
  • function 支持已匿名方式传参
-- --------
-- 匿名函数
-- --------
function testFunc(tab, fn)
    for k,v in pairs(tab) do
        print(fn(k,v))
    end
end

tab1 = {key1="value1",key2="value2"}
testFunc(tab1,
function(a,b)
    return a .. " - " .. b
end)
